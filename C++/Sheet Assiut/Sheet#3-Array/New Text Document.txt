#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,i,cont=0,mini;
    cin>>n;
    int arr[n];
    for(i=0;i<n;i++)
        cin>>arr[i];
    sort(arr,arr+n);
    mini=arr[0];
    for(i=0;i<n;i++)
        if(arr[i]==mini)
            {
            cont++;
            }
    if(cont%2==0){
        cout<<"Unlucky"<<endl;
    }else{
    cout<<"Lucky"<<endl;
    }
    return 0;
}
